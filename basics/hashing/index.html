sort(vector<int> a, int k):
    n = a.size()
    if n >= 2 ** (2 ** k):
        countingSort(a)
        return
    vector<int> l # —Å—é–¥–∞ –∑–∞–ø–∏—à–µ–º –≤—Å–µ –≤—Å—Ç—Ä–µ—Ç–∏–≤—à–∏–µ—Å—è —Å—Ç–∞—Ä—à–∏–µ —Ü–∏—Ñ—Ä—ã

    # –∞—Å—Å–æ—Ü–∏–∞—Ç–∏–≤–Ω—ã–π –º–∞—Å—Å–∏–≤: –≤ t[x] –±—É–¥–µ–º —Ö—Ä–∞–Ω–∏—Ç—å —Å–ø–∏—Å–æ–∫ –≤—Å–µ—Ö —Ç–∞–∫–∏—Ö —É,
    # —á—Ç–æ a[i] = x * 2 ** (2 ** (k - 1)) + y –¥–ª—è –Ω–µ–∫–æ—Ç–æ—Ä–æ–≥–æ i
    for i = 0..(n - 1):
        x = a[i] / (2 ** (2 ** (k - 1))) # x - —Å—Ç–∞—Ä—à–∏–µ 2 ** (k - 1) 6–∏—Ç a[i]
        y = a[i] % (2 ** (2 ** (k - 1))) # y - –º–ª–∞–¥—à–∏–∏–µ 2 ** (k - 1) –±–∏—Ç –∞[i]
        if t.find(x) == t.end(): # x –≤—Å—Ç—Ä–µ—Ç–∏–ª–æ—Å—å –≤–ø–µ—Ä–≤—ã–µ
            l.push_back(x)
        t[x].push_back(y)
    sort(l, k - 1) # –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–ª–∏ —Å—Ç–∞—Ä—à–∏–µ —Ü–∏—Ñ—Ä—ã
    a.clear()
    for (x in l): # –ø–µ—Ä–µ–±–∏—Ä–∞–µ–º —Å—Ç–∞—Ä—à–∏–µ —Ü–∏—Ñ—Ä—ã –≤ –ø–æ—Ä—è–¥–∫–µ –≤–æ–∑—Ä–∞—Å—Ç–∞–Ω–∏—è
        r = t[x]
        i = maxElementIndex(r) # –Ω–∞–π–¥—ë–º –∏–Ω–¥–µ–∫—Å –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞ –≤ r
        rmax = r[i]
        swap(r[i], r.back())
        r.pop_back()
        sort(r, k - 1) # –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–ª–∏ –≤—Å–µ –º–ª–∞–¥—à–∏–µ —Ü–∏—Ñ—Ä—ã, –∫—Ä–æ–º–µ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–π
        for (y in r):
            a.push_back(x * 2 ** (2 ** (k - 1)) + y)
        a.push_back(x * 2 ** (2 ** (k - 1)) + rmax)

–î–æ–∫–∞–∑–∞—Ç–µ–ª—å—Å—Ç–≤–æ

–î–ª—è –∫–∞–∂–¥–æ–π –ø–∞—Ä—ã –∫–ª—é—á–µ–π $a, b \in U$ –≤–≤–µ–¥—ë–º –≤–µ–ª–∏—á–∏–Ω—É $\chi(h, a, b)$, —Ä–∞–≤–Ω—É—é –µ–¥–∏–Ω–∏—Ü–µ, –µ—Å–ª–∏ $h(a)=h(b)$, –∏ –Ω—É–ª—é –∏–Ω–∞—á–µ. $\chi(h, a, a)=1$ –¥–ª—è –ª—é–±–æ–π $h$. –ü—Ä–∏ —ç—Ç–æ–º –¥–ª—è $a \neq b$ –º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ $\chi(h, a, b)$ –Ω–µ –ø—Ä–µ–≤–æ—Å—Ö–æ–¥–∏—Ç $\frac{1}{m}$ –ø–æ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—é —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–æ–≥–æ —Å–µ–º–µ–π—Å—Ç–≤–∞:

$$ \mathbb{E}(\chi(h, a, b))=\frac{1}{|\mathcal{H}|} \sum_{h \in \mathcal{H}} \chi(h, a, b) \leqslant \frac{1}{|\mathcal{H}|} \cdot \frac{|\mathcal{H}|}{m}=\frac{1}{m} $$

–û–±–æ–∑–Ω–∞—á–∏–º $L=\left\{l_{1}, \ldots, l_{n}\right\}$. –ú–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ –¥–ª–∏–Ω—ã —Å–ø–∏—Å–∫–∞ –≤ —è—á–µ–π–∫–µ —Å –∏–Ω–¥–µ–∫—Å–æ–º $h(k)$ —Ä–∞–≤–Ω—è–µ—Ç—Å—è

$$ \mathbb{E}\left(\sum_{l \in L} \chi(h, k, l)\right)=\sum_{l \in L} \mathbb{E}(\chi(h, k, l)) \leqslant 1+\sum_{l \in L \backslash\{k\}} \mathbb{E}(\chi(h, k, l)) \leqslant 1+\frac{n}{m} $$

–í –ª—é–±–æ–π –º–æ–º–µ–Ω—Ç –≤—Ä–µ–º–µ–Ω–∏ –≤ —Ç–∞–±–ª–∏—Ü–µ –ª–µ–∂–∞—Ç –Ω–µ –±–æ–ª–µ–µ $O(m)$ —ç–ª–µ–º–µ–Ω—Ç–æ–≤. –í—Ä–µ–º—è —Ä–∞–±–æ—Ç—ã –ª—é–±–æ–π –æ–ø–µ—Ä–∞—Ü–∏–∏ –Ω–µ –ø—Ä–µ–≤–æ—Å—Ö–æ–¥–∏—Ç –≤—Ä–µ–º–µ–Ω–∏ –≤—ã—á–∏—Å–ª–µ–Ω–∏—è —Ö–µ—à-—Ñ—É–Ω–∫—Ü–∏–∏ (–∫–æ—Ç–æ—Ä–æ–µ –º—ã —Å—á–∏—Ç–∞–µ–º –∫–æ–Ω—Å—Ç–∞–Ω—Ç–Ω—ã–º) –∏ –≤—Ä–µ–º–µ–Ω–∏ –ø—Ä–æ—Ö–æ–¥–∞ –ø–æ —Å–ø–∏—Å–∫—É –≤ —è—á–µ–π–∫–µ —Å –∏–Ω–¥–µ–∫—Å–æ–º, —Ä–∞–≤–Ω—ã–º –∑–Ω–∞—á–µ–Ω–∏—é —Ö–µ—à-—Ñ—É–Ω–∫—Ü–∏–∏. –ü–æ –ø—Ä–µ–¥—ã–¥—É—â–µ–π —Ç–µ–æ—Ä–µ–º–µ, –º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ –≤—Ä–µ–º–µ–Ω–∏ –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è –∫–∞–∂–¥–æ–π –æ–ø–µ—Ä–∞—Ü–∏–∏ –Ω–µ –ø—Ä–µ–≤–æ—Å—Ö–æ–¥–∏—Ç $O\left(1+\frac{O(m)}{m}\right)=O(1)$. –¢–æ–≥–¥–∞ –º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ —Å—É–º–º–∞—Ä–Ω–æ–≥–æ –≤—Ä–µ–º–µ–Ω–∏ —Ä–∞–±–æ—Ç—ã –≤—Å–µ—Ö –æ–ø–µ—Ä–∞—Ü–∏–π –Ω–µ –ø—Ä–µ–≤–æ—Å—Ö–æ–¥–∏—Ç $O(k)$, —Ç–æ –µ—Å—Ç—å —Ä–∞–≤–Ω—è–µ—Ç—Å—è $\Theta(k)$.

–ï—â—ë —Ä–∞–∑ –æ—Ç–º–µ—Ç–∏–º, —á—Ç–æ —Ç–µ–æ—Ä–µ–º–∞ –∏ —Å–ª–µ–¥—Å—Ç–≤–∏–µ –≤—ã–ø–æ–ª–Ω—è—é—Ç—Å—è –¥–ª—è –ª—é–±—ã—Ö –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–µ–π –∑–∞–ø—Ä–æ—Å–æ–≤; –º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ –±–µ—Ä—ë—Ç—Å—è –Ω–µ –ø–æ –≤—Ö–æ–¥–Ω—ã–º –¥–∞–Ω–Ω—ã–º, –∞ –ø–æ —Å–ª—É—á–∞–π–Ω–æ–º—É –≤—ã–±–æ—Ä—É —Ö–µ—à-—Ñ—É–Ω–∫—Ü–∏–∏.

–ü–æ—Å—Ç—Ä–æ–µ–Ω–∏–µ —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–æ–≥–æ —Å–µ–º–µ–π—Å—Ç–≤–∞ —Ö–µ—à-—Ñ—É–Ω–∫—Ü–∏–π

–ü–æ—Å—Ç—Ä–æ–∏–º —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–æ–µ —Å–µ–º–µ–π—Å—Ç–≤–æ —Ö–µ—à-—Ñ—É–Ω–∫—Ü–∏–π –¥–ª—è —á–∏—Å–µ–ª, –≤–ª–µ–∑–∞—é—â–∏—Ö –≤ –º–∞—à–∏–Ω–Ω–æ–µ —Å–ª–æ–≤–æ (—Å—á–∏—Ç–∞–µ–º, —á—Ç–æ –∞—Ä–∏—Ñ–º–µ—Ç–∏—á–µ—Å–∫–∏–µ –æ–ø–µ—Ä–∞—Ü–∏–∏ –Ω–∞–¥ —á–∏—Å–ª–∞–º–∏ –≤—ã–ø–æ–ª–Ω—è—é—Ç—Å—è –∑–∞ $O(1)$ ).

–¢–µ–æ—Ä–µ–º–∞

–ü—É—Å—Ç—å $p$ - —Ç–∞–∫–æ–µ –ø—Ä–æ—Å—Ç–æ–µ, —á—Ç–æ $U \subset\{0,1, \ldots, p-1\}$; –ø—É—Å—Ç—å $m < p$. –î–ª—è –ª—é–±—ã—Ö —Ü–µ–ª—ã—Ö $1 \leqslant a < p, 0 \leqslant b < p$ –æ–ø—Ä–µ–¥–µ–ª–∏–º —Ö–µ—à-—Ñ—É–Ω–∫—Ü–∏—é

$$ h_{a, b}(k)=((a k+b) \bmod p) \bmod m $$

–¢–æ–≥–¥–∞ —Å–µ–º–µ–π—Å—Ç–≤–æ —Ö–µ—à-—Ñ—É–Ω–∫—Ü–∏–π

$$ \mathcal{H}_{p, m}=\left\{h_{a, b}: 1 \leqslant a < p, 0 \leqslant b < p\right\} $$

—è–≤–ª—è–µ—Ç—Å—è —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–º.

–î–æ–∫–∞–∑–∞—Ç–µ–ª—å—Å—Ç–≤–æ

–ü—É—Å—Ç—å $k \neq l$ - –¥–≤–∞ —Ä–∞–∑–ª–∏—á–Ω—ã—Ö –∫–ª—é—á–∞. –ü–æ–π–º—ë–º, –¥–ª—è –∫–∞–∫–∏—Ö $a, b$ –≤–µ—Ä–Ω–æ $h_{a, b}(k)=h_{a, b}(l)$.

–ó–∞–º–µ—Ç–∏–º —Å–Ω–∞—á–∞–ª–∞, —á—Ç–æ –µ—Å–ª–∏ $r=(a k+b) \bmod p, s=(a l+b) \bmod p$, —Ç–æ $r \neq s$, —Ç–∞–∫ –∫–∞–∫ $r-s \equiv a(k-l) \not \equiv 0(\bmod p)$.

–ë–æ–ª–µ–µ —Ç–æ–≥–æ, —Ä–∞–∑–Ω—ã–º –ø–∞—Ä–∞–º $(a, b)$ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—Ç —Ä–∞–∑–Ω—ã–µ –ø–∞—Ä—ã $(r, s)$, —Ç–∞–∫ –∫–∞–∫ –ø–æ $(r, s)$ –º–æ–∂–Ω–æ –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–∏—Ç—å $(a, b): a \equiv(r-s)(k-l)^{-1}(\bmod p), b \equiv(r-a k)(\bmod p)$.

–†–∞–∑–ª–∏—á–Ω—ã—Ö –ø–∞—Ä $(a, b)$ —Å $1 \leqslant a < p, 0 \leqslant b < p$ –≤—Å–µ–≥–æ $(p-1) \cdot p$. –†–∞–∑–ª–∏—á–Ω—ã—Ö –ø–∞—Ä $(r, s)$ —Å $0 \leqslant r \neq s < p$ –≤—Å–µ–≥–æ $p^{2}-p$, —Ç–æ –µ—Å—Ç—å —Å—Ç–æ–ª—å–∫–æ –∂–µ. –ó–Ω–∞—á–∏—Ç, –º—ã —É—Å—Ç–∞–Ω–æ–≤–∏–ª–∏ –≤–∑–∞–∏–º–Ω–æ –æ–¥–Ω–æ–∑–Ω–∞—á–Ω–æ–µ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ –º–µ–∂–¥—É –º–Ω–æ–∂–µ—Å—Ç–≤–∞–º–∏ –ø–∞—Ä ( $a, b$ ) –∏ –ø–∞—Ä $(r, s)$.

–û—Å—Ç–∞—ë—Ç—Å—è –∑–∞–º–µ—Ç–∏—Ç—å, —á—Ç–æ $h_{a, b}(k)=h_{a, b}(l)$ —Ç–æ–≥–¥–∞ –∏ —Ç–æ–ª—å–∫–æ —Ç–æ–≥–¥–∞, –∫–æ–≥–¥–∞ $r \equiv s(\bmod m)$. –î–ª—è —Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ $0 \leqslant r < p$ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ $0 \leqslant s < p$ —Ç–∞–∫–∏—Ö, —á—Ç–æ $s \neq r$, –Ω–æ $r \equiv s(\bmod m)$, –Ω–µ –ø—Ä–µ–≤–æ—Å—Ö–æ–¥–∏—Ç $\left\lceil\frac{p}{m}\right\rceil-1 \leqslant \frac{p+m-1}{m}-1=\frac{p-1}{m}$.

–¢–æ–≥–¥–∞ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ö–µ—à-—Ñ—É–Ω–∫—Ü–∏–π $h_{a, b} \in \mathcal{H}_{p, m}$, –¥–ª—è –∫–æ—Ç–æ—Ä—ã—Ö $h_{a, b}(k)=h_{a, b}(l)$ —Ä–∞–≤–Ω—è–µ—Ç—Å—è –∫–æ–ª–∏—á–µ—Å—Ç–≤—É –ø–∞—Ä $(r, s)$ —Ç–∞–∫–∏—Ö, —á—Ç–æ $0 \leqslant r \neq s < p, r \equiv s(\bmod m)$, –∫–æ—Ç–æ—Ä–æ–µ –Ω–µ –ø—Ä–µ–≤–æ—Å—Ö–æ–¥–∏—Ç $p \cdot \frac{p-1}{m}=\frac{\left|\mathcal{H}_{p, m}\right|}{m}$.

–°–æ–≤–µ—Ä—à–µ–Ω–Ω–æ–µ —Ö–µ—à–∏—Ä–æ–≤–∞–Ω–∏–µ

–ï—Å–ª–∏ –º–Ω–æ–∂–µ—Å—Ç–≤–æ –∏—Å–ø–æ–ª—å–∑—É–µ–º—ã—Ö –∫–ª—é—á–µ–π –∏–∑–≤–µ—Å—Ç–Ω–æ –∑–∞—Ä–∞–Ω–µ–µ (–Ω–∞–ø—Ä–∏–º–µ—Ä, –≤ —è–∑—ã–∫–∞—Ö –ø—Ä–æ–≥—Ä–∞–º–º–∏—Ä–æ–≤–∞–Ω–∏—è –º–Ω–æ–∂–µ—Å—Ç–≤–æ –∑–∞—Ä–µ–∑–µ—Ä–≤–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö —Å–ª–æ–≤ —Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–æ), —Ç–æ –º–æ–∂–Ω–æ –ø–æ—Å—Ç—Ä–æ–∏—Ç—å —Ö–µ—à-—Ç–∞–±–ª–∏—Ü—É, –æ–ø–µ—Ä–∞—Ü–∏–∏ –≤ –∫–æ—Ç–æ—Ä–æ–π –±—É–¥—É—Ç —Ä–∞–±–æ—Ç–∞—Ç—å –∑–∞ $O(1)$ –≤ —Ö—É–¥—à–µ–º —Å–ª—É—á–∞–µ. –ü—Ä–∏ —ç—Ç–æ–º –º–æ–∂–Ω–æ –¥–æ–±–∏—Ç—å—Å—è —Ç–æ–≥–æ, —á—Ç–æ–±—ã —Ö–µ—à-—Ç–∞–±–ª–∏—Ü–∞ –∏–º–µ–ª–∞ —Ä–∞–∑–º–µ—Ä $O(n)$, –≥–¥–µ $n$ - —á–∏—Å–ª–æ –∏—Å–ø–æ–ª—å–∑—É–µ–º—ã—Ö –∫–ª—é—á–µ–π.

–°–Ω–æ–≤–∞ —Å—á–∏—Ç–∞–µ–º, —á—Ç–æ –≤—Å–µ –∫–ª—é—á–∏ - —Ü–µ–ª—ã–µ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–µ —á–∏—Å–ª–∞, –º–µ–Ω—å—à–∏–µ –Ω–µ–∫–æ—Ç–æ—Ä–æ–≥–æ –ø—Ä–æ—Å—Ç–æ–≥–æ $p$. –ü–æ—Å—Ç—Ä–æ–∏–º —Ö–µ—à-—Ç–∞–±–ª–∏—Ü—É —Å $m=n$ —è—á–µ–π–∫–∞–º–∏, –ø—Ä–∏ —ç—Ç–æ–º —Ö–µ—à-—Ñ—É–Ω–∫—Ü–∏—é –∞–∫–∫—É—Ä–∞—Ç–Ω–æ –≤—ã–±–µ—Ä–µ–º –∏–∑ —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–æ–≥–æ —Å–µ–º–µ–π—Å—Ç–≤–∞ $\mathcal{H}_{p, m}$. –í –∫–∞–∂–¥–æ–π —è—á–µ–π–∫–µ —ç—Ç–æ–π —Ç–∞–±–ª–∏—Ü—ã –≤–º–µ—Å—Ç–æ —Å–ø–∏—Å–∫–∞ –º—ã –∑–∞–≤–µ–¥—ë–º —Ö–µ—à-—Ç–∞–±–ª–∏—Ü—É –≤—Ç–æ—Ä–æ–≥–æ —É—Ä–æ–≤–Ω—è, –ø—Ä–∏—á—ë–º, –µ—Å–ª–∏ –≤ —è—á–µ–π–∫—É —Å –∏–Ω–¥–µ–∫—Å–æ–º $j$ –ø–æ–ø–∞–ª–æ $n_{j}$ –∫–ª—é—á–µ–π, —Ç–æ –≤–Ω—É—Ç—Ä–µ–Ω–Ω—é—é —Ö–µ—à-—Ç–∞–±–ª–∏—Ü—É –≤ —ç—Ç–æ–π —è—á–µ–π–∫–µ —Å–¥–µ–ª–∞–µ–º —Ä–∞–∑–º–µ—Ä–∞ $m_{j}=n_{j}^{2}$. –•–µ—à—Ñ—É–Ω–∫—Ü–∏—é –¥–ª—è –≤–Ω—É—Ç—Ä–µ–Ω–Ω–µ–π —Ç–∞–±–ª–∏—Ü—ã –º—ã –∞–∫–∫—É—Ä–∞—Ç–Ω–æ –≤—ã–±–µ—Ä–µ–º –∏–∑ —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–æ–≥–æ —Å–µ–º–µ–π—Å—Ç–≤–∞ $\mathcal{H}_{p, m_{j}}$.

–û–∫–∞–∑—ã–≤–∞–µ—Ç—Å—è, –º–æ–∂–Ω–æ –ø–æ–¥–æ–±—Ä–∞—Ç—å —Ç–∞–∫–∏–µ —Ö–µ—à-—Ñ—É–Ω–∫—Ü–∏–∏, —á—Ç–æ –≤–æ –≤–Ω—É—Ç—Ä–µ–Ω–Ω–∏—Ö —Ç–∞–±–ª–∏—Ü–∞—Ö –Ω–µ –±—É–¥–µ—Ç –∫–æ–ª–ª–∏–∑–∏–π, –∞ —Å—É–º–º–∞—Ä–Ω—ã–π —Ä–∞–∑–º–µ—Ä –≤—Å–µ—Ö —Ç–∞–±–ª–∏—Ü –±—É–¥–µ—Ç –æ—Ü–µ–Ω–∏–≤–∞—Ç—å—Å—è –∫–∞–∫ $O(n)$.

–í –¥–æ–∫–∞–∑–∞—Ç–µ–ª—å—Å—Ç–≤–µ –æ—Ü–µ–Ω–æ–∫ –º—ã –±—É–¥–µ–º –ø–æ–ª—å–∑–æ–≤–∞—Ç—å—Å—è —Å–ª–µ–¥—É—é—â–∏–º —Ñ–∞–∫—Ç–æ–º (–ø–æ–¥ $\mathbb{P}(A)$ –∏–º–µ–µ—Ç—Å—è –≤ –≤–∏–¥—É –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å —Ç–æ–≥–æ, —á—Ç–æ –ø—Ä–æ–∏–∑–æ—à–ª–æ —Å–æ–±—ã—Ç–∏–µ $A$ ):

–ù–µ—Ä–∞–≤–µ–Ω—Å—Ç–≤–æ –ú–∞—Ä–∫–æ–≤–∞

–ü—É—Å—Ç—å $X$ - –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞—è —Å–ª—É—á–∞–π–Ω–∞—è –≤–µ–ª–∏—á–∏–Ω–∞ —Å –∫–æ–Ω–µ—á–Ω—ã–º –º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–∏–º –æ–∂–∏–¥–∞–Ω–∏–µ–º. –¢–æ–≥–¥–∞ –¥–ª—è –ª—é–±–æ–≥–æ $a>0$ –≤–µ—Ä–Ω–æ

$$ \mathbb{P}(X \geqslant a) \leqslant \frac{\mathbb{E} X}{a} $$
–î–æ–∫–∞–∑–∞—Ç–µ–ª—å—Å—Ç–≤–æ

–ü–æ—Å–∫–æ–ª—å–∫—É $X$ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–æ, $\mathbb{E} X \geqslant 0 \cdot \mathbb{P}(X < a)+a \cdot \mathbb{P}(X \geqslant a)$, –æ—Ç–∫—É–¥–∞ —Å–ª–µ–¥—É–µ—Ç —Ç—Ä–µ–±—É–µ–º–æ–µ –Ω–µ—Ä–∞–≤–µ–Ω—Å—Ç–≤–æ.

–¢–µ–æ—Ä–µ–º–∞

–ü—É—Å—Ç—å –≤ —Ö–µ—à-—Ç–∞–±–ª–∏—Ü–µ —Ä–∞–∑–º–µ—Ä–∞ $m=n^{2}$ —Ö—Ä–∞–Ω—è—Ç—Å—è $n$ –∫–ª—é—á–µ–π, –ø—Ä–∏—á—ë–º —Ö–µ—à—Ñ—É–Ω–∫—Ü–∏—è $h(\cdot)$ –±—ã–ª–∞ —Å–ª—É—á–∞–π–Ω–æ –≤—ã–±—Ä–∞–Ω–∞ –∏–∑ $\mathcal{H}_{p, m}$. –¢–æ–≥–¥–∞ —Å –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å—é –±–æ–ª–µ–µ —á–µ–º $\frac{1}{2}$ –∫–æ–ª–ª–∏–∑–∏–π –Ω–µ—Ç (–≤—Å–µ –∫–ª—é—á–∏ —Ö—Ä–∞–Ω—è—Ç—Å—è –≤ —Ä–∞–∑–Ω—ã—Ö —è—á–µ–π–∫–∞—Ö).

–î–æ–∫–∞–∑–∞—Ç–µ–ª—å—Å—Ç–≤–æ

–î–ª—è –∫–∞–∂–¥–æ–π –ø–∞—Ä—ã —Ä–∞–∑–ª–∏—á–Ω—ã—Ö –∫–ª—é—á–µ–π –≤–≤–µ–¥—ë–º –≤–µ–ª–∏—á–∏–Ω—É $\chi(h, a, b)$, —Ä–∞–≤–Ω—É—é –µ–¥–∏–Ω–∏—Ü–µ, –µ—Å–ª–∏ $h(a)=h(b)$, –∏ –Ω—É–ª—é –∏–Ω–∞—á–µ. –ö–∞–∫ –∏ –≤ —Ç–µ–æ—Ä–µ–º–µ –£–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–æ–≥–æ —Ö–µ—à–∏—Ä–æ–≤–∞–Ω–∏—è 8.6.1, $\mathbb{E}(\chi(h, a, b)) \leqslant \frac{1}{m}=\frac{1}{n^{2}}$. –†–∞—Å—Å–º–æ—Ç—Ä–∏–º —Å–ª—É—á–∞–π–Ω—É—é –≤–µ–ª–∏—á–∏–Ω—É $X$ - –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∫–æ–ª–ª–∏–∑–∏–π. –¢–æ–≥–¥–∞

$$ \mathbb{E} X=\mathbb{E}\left(\sum_{a \neq b}(\chi(h, a, b))\right)=\sum_{a \neq b} \mathbb{E}(\chi(h, a, b)) \leqslant \frac{n(n-1)}{2} \cdot \frac{1}{n^{2}}<\frac{1}{2} $$
