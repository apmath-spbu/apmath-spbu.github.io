hirschberg(a, b):

    if n <= 1 or m <= 1:
        # –µ—Å–ª–∏ –æ–¥–Ω–∞ –∏–∑ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–µ–π –∏–º–µ–µ—Ç –¥–ª–∏–Ω—É –Ω–µ –±–æ–ª—å—à–µ 1,

int l[2][m + 1]
int mid[2][m + 1]
relax(i, j, newL, newMid):
    if l[i, j] < newL:
        l[i, j] = newL, mid[i, j] = newMid
hirschberg(a, b):
    n = len(a), m = len(b)
    if n <= 1 or m <= 1:
        return naiveLCS(a, b)
    fill(l, 0), fill(mid, 0)
    for i = 1..n:
        cur = i % 2, prev = 1 - cur
        for j = 1..m:
            relax(cur, j, l[prev, j], mid[prev, j])
            relax(cur, j, l[cur, j - 1], mid[cur, j - 1])
            if a[i] == b[j]:
                relax(cur, j, l[prev, j - 1] + 1, mid[prev, j - 1])
            if i == n / 2:
                mid[cur, j] = j
    j = mid[n, m]
    return hirschberg(a[0, n / 2), b[0, j))
            + hirschberg(a[n / 2, n), b[j, m))
