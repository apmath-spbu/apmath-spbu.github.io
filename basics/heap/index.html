        largest = left

void siftDown(std::vector<int>& a, int i, int n) {
    int largest = i;
    int left = 2 * i + 1;  // –ª–µ–≤—ã–π –ø–æ—Ç–æ–º–æ–∫ (–≤ 0-–∏–Ω–¥–µ–∫—Å–∞—Ü–∏–∏)

    // –ï—Å–ª–∏ –ª–µ–≤—ã–π –ø–æ—Ç–æ–º–æ–∫ –±–æ–ª—å—à–µ –∫–æ—Ä–Ω—è
    if (left < n && a[left] > a[largest]) {
        largest = left;
    }
    
    // –ï—Å–ª–∏ –ø—Ä–∞–≤—ã–π –ø–æ—Ç–æ–º–æ–∫ –±–æ–ª—å—à–µ, —á–µ–º —Å–∞–º—ã–π –±–æ–ª—å—à–æ–π —ç–ª–µ–º–µ–Ω—Ç –Ω–∞ –¥–∞–Ω–Ω—ã–π –º–æ–º–µ–Ω—Ç
    if (right < n && a[right] > a[largest]) {
        largest = right;
    }
    
    // –ï—Å–ª–∏ —Å–∞–º—ã–π –±–æ–ª—å—à–æ–π —ç–ª–µ–º–µ–Ω—Ç –Ω–µ –∫–æ—Ä–µ–Ω—å
    if (largest != i) {
        std::swap(a[i], a[largest]);  // –º–µ–Ω—è–µ–º –º–µ—Å—Ç–∞–º–∏
        // –†–µ–∫—É—Ä—Å–∏–≤–Ω–æ –ø—Ä–æ—Å–µ–∏–≤–∞–µ–º –∑–∞—Ç—Ä–æ–Ω—É—Ç—É—é –ø–æ–¥–∫—É—á—É
        siftDown(a, largest, n);
    }
}

void build(std::vector<int>& a, int n) {
    // –ù–∞—á–∏–Ω–∞–µ–º —Å –ø–æ—Å–ª–µ–¥–Ω–µ–≥–æ —Ä–æ–¥–∏—Ç–µ–ª—è –∏ –∏–¥–µ–º –∫ –∫–æ—Ä–Ω—é
